        \hlcom{# on the test x, by normalization}

                \hlstd{fit}\hlopt{$}\hlstd{chisq}
        \hlstd{\},} \hlnum{1}\hlopt{:}\hlkwd{length}\hlstd{(val.test.threshold), val.test.threshold)}
